114void calc_accelleration(const size_t particle_index) {
115 vec3 sum = {0, 0, 0};
116
117 const vec3 pos = points[particle_index];
118 for(int i = 0; i < particles; i++)
119 if(i != particle_index) {
120 const double scalar = pirogi2[particle_index] + pirogi2[i];
121 sum = sum + (gradW(pos - points[i]) * scalar);
122 }
123
124 accelerations[particle_index] = (sum * -m) + (pos * -lambda) + (velocities[particle_index] * -nu);
