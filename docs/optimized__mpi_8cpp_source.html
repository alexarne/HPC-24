48void init_tree_sizes() {
49 MPI_Status status;

51 MPI_Recv(self_node.range, 2, MPI_INT, parent_rank, SEND_REGIONS, MPI_COMM_WORLD, &status);
52
53 if(children.size() == 0) return;

56 if(children.size() == 2)

59 partition_points.push_back(self_node.right());

61 for(int i = 0; i < children.size(); i++) {
